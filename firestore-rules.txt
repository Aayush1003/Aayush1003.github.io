Firestore Security Rules Template
// Go to Firebase Console > Firestore Database > Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow read/write access to contacts collection
    match /contacts/{document} {
      allow create: if request.auth != null || true; // Allow anonymous writes for contact form
      allow read, update, delete: if request.auth != null;
    }

    // Allow read access to other collections if needed
    match /{document=**} {
      allow read: if true;
      allow write: if request.auth != null;
    }
  }
}

// Note: For production, consider restricting anonymous writes
// and implementing proper authentication
